=== ENVIRONMENT ==========
$SHELL					// current shell
cat /etc/shells			// list of available shells
$PS1					// current prompt customization
$PWD					// current directory
$OLDPWD					// previous directory
$PATH					// paths
$HOME					// current user home directory
$USER					// current user name

=== SPECIAL VARIABLES ====
$$						// process ID
$?						// exit status of last command (0 if no error)
$!						// process ID of last background job
$RANDOM					// random number

#!/bin/bash				// fisrt line in a shell script : specify to the OS which shell to use (called the shebang line)

=== PERMISSIONS ==========
chmod +x script.sh		// add execution permission to everyone for script.sh
chmod 700 script.sh 	// add all permissions to owner and none to everybody else
permission order : owner, group, other | read, write, execute

=== MULTI COMMANDS =======
command1; command2		// command2 runs regardless of command1 success
command1 && command2	// command2 runs only if command1 succeeds
command1 || command2	// command2 runs only if command1 fails
\						// For multiline commands, long commands needing to be cut by continuing on another line

=== REDIRECTION ==========
>						// This takes the output from the preceding command that youâ€™d normally see in the terminal and sends it to a file that you give it
ex: pip freeze > requirements.txt
<						// This takes the content of a file and sends it as input to a command
ex: pbcopy < ~/Desktop/folder/file.txt
>>						// Append to file : will not overwrite
|						// A pipe takes the standard output of one command and passes it as the input to another.
ex: wc -l file1.txt|tee -a file2.txt
